Javascript两种function定义的区别




Javascript有两种常用定义Function的方法

	 function a(){alert("a")}

	 var a = function(){alert("a")}

Javascript脚本会优先加载function fn(){}，变量方式声明则按顺序。

	b();
	a();
	

	var a = function() {
   		alert('a');
	}

	function b() {
    		alert('b');
	}
	

	
	// b
	// a is not a function

同时，其中一个是匿名函数

	var a = function() {
    		alert('a');
	}

	function b() {
    		alert('b');
	}
	

	alert(a.prototype.constructor);

	alert(b.prototype.constructor);
	

	
	function() {
    		alert('a');
	}

	function b() {
    		alert('b');
